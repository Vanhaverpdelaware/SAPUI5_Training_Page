With @sapui5/ts-types-esm installed, VS Code gives you full autocompletion and type checking for all UI5 controls and APIs.

Importing UI5 Modules

Use ES module imports — ui5-tooling-transpile converts these to sap.ui.define() at build time:

import Controller from "sap/ui/core/mvc/Controller";
import JSONModel from "sap/ui/model/json/JSONModel";
import MessageBox from "sap/m/MessageBox";
import MessageToast from "sap/m/MessageToast";
import Button from "sap/m/Button";
import Event from "sap/ui/base/Event";

TypeScript Controller Pattern

The standard controller pattern at delaware (from QA-003):

import Controller from "sap/ui/core/mvc/Controller";
import JSONModel from "sap/ui/model/json/JSONModel";
import ODataModel from "sap/ui/model/odata/v4/ODataModel";
import Event from "sap/ui/base/Event";

/** @namespace myapp.controller */
export default class Selection extends Controller {
  private _oViewModel!: JSONModel;
  private _oDataModel!: ODataModel;

  public onInit(): void {
    this._oViewModel = new JSONModel({
      plant: "",
      isValid: false,
      busy: false
    });
    this.getView()?.setModel(this._oViewModel, "view");
    this._oDataModel = this.getView()?.getModel() as ODataModel;
  }

  public onExecutePress(): void {
    this._executeAnalysis();
  }

  public onPlantChange(oEvent: Event): void {
    const sValue = (oEvent.getParameter("value") as string).trim();
    this._oViewModel.setProperty("/plant", sValue);
    this._oViewModel.setProperty("/isValid", sValue.length === 4);
  }

  private _executeAnalysis(): void {
    if (!this._oViewModel.getProperty("/isValid") as boolean) {
      MessageBox.error("Plant is required");
      return;
    }
    this._oViewModel.setProperty("/busy", true);
    // ... call OData service
  }
}

TypeScript Classes for Reusable Logic

Extract reusable logic into separate TypeScript classes — do not put everything in controllers. Example from the delaware class-based development guide:

import JSONModel from "sap/ui/model/json/JSONModel";

type WorkcenterType = { id: string; name: string; };

export default class WorkcenterService {
  private model: JSONModel;

  constructor(model: JSONModel) {
    super();
    this.model = model;
  }

  set workcenters(data: WorkcenterType[]) {
    this.model.setProperty("/Workcenters", data);
  }

  get workcenters(): WorkcenterType[] {
    return this.model.getProperty("/Workcenters") as WorkcenterType[];
  }
}

Then use it in your controller:

import WorkcenterService from "../model/WorkcenterService";

private _service!: WorkcenterService;

public onInit(): void {
  const oModel = new JSONModel();
  this.getView()?.setModel(oModel);
  this._service = new WorkcenterService(oModel);
}

Error Handling with Types

import MessageBox from "sap/m/MessageBox";
import MessageToast from "sap/m/MessageToast";

private _showError(sMessage: string): void {
  MessageBox.error(sMessage, {
    title: this._getText("errorTitle")
  });
}

private _showSuccess(sMessage: string): void {
  MessageToast.show(sMessage);
}

private _getText(sKey: string): string {
  const oBundle = this.getView()
    ?.getModel("i18n")
    ?.getResourceBundle() as ResourceBundle;
  return oBundle?.getText(sKey) ?? sKey;
}
