Exercise 2

Building on Exercise 1, you add routing so clicking a list item navigates to a detail view — the standard Fiori master-detail pattern.

Prerequisites: Complete Exercise 1 before starting.

Goal

Add a second view and configure routing so that clicking an item in the list navigates to a detail page showing that item's data.

Step 1 — Create the Detail view

Create webapp/view/Detail.view.xml:

<mvc:View controllerName="ui5.training.controller.Detail"
  xmlns="sap.m" xmlns:mvc="sap.ui.core.mvc">
  <Page
    title="{name}"
    showNavButton="true"
    navButtonPress=".onNavBack">
    <content>
      <VBox class="sapUiMediumMargin">
        <Label text="Material ID" />
        <Text text="{id}" />
        <Label text="Name" class="sapUiSmallMarginTop" />
        <Text text="{name}" />
        <Label text="Quantity" class="sapUiSmallMarginTop" />
        <Text text="{quantity}" />
      </VBox>
    </content>
  </Page>
</mvc:View>

Step 2 — Create the Detail controller

Create webapp/controller/Detail.controller.ts:

import Controller from "sap/ui/core/mvc/Controller";
import { History } from "sap/ui/core/routing/History";

/** @namespace ui5.training.controller */
export default class Detail extends Controller {
  public onNavBack(): void {
    const oHistory = History.getInstance();
    const sPreviousHash = oHistory.getPreviousHash();

    if (sPreviousHash !== undefined) {
      window.history.go(-1);
    } else {
      this.getOwnerComponent()?.getRouter()
        .navTo("main", {}, true);
    }
  }
}

Step 3 — Configure routing in manifest.json

Add routes and targets to sap.ui5.routing in manifest.json:

"routing": {
  "config": {
    "routerClass": "sap.m.routing.Router",
    "viewType": "XML",
    "viewPath": "ui5.training.view",
    "controlId": "app",
    "controlAggregation": "pages",
    "async": true
  },
  "routes": [
    { "name": "main",   "pattern": "",           "target": "main" },
    { "name": "detail", "pattern": "detail/{id}", "target": "detail" }
  ],
  "targets": {
    "main":   { "viewName": "Main",   "viewLevel": 1 },
    "detail": { "viewName": "Detail", "viewLevel": 2 }
  }
}

Step 4 — Navigate on item press

In Main.view.xml, add a press handler to each list item:

<StandardListItem
  title="{name}"
  description="Qty: {quantity}"
  type="Navigation"
  press=".onItemPress" />

In Main.controller.ts, add the handler:

import Event from "sap/ui/base/Event";
import ListItem from "sap/m/ListItemBase";

public onItemPress(oEvent: Event): void {
  const oItem = oEvent.getSource() as ListItem;
  const oCtx = oItem.getBindingContext();
  const sId = oCtx?.getProperty("id") as string;

  this.getOwnerComponent()?.getRouter()
    .navTo("detail", { id: sId });
}

Step 5 — Bind data in Detail controller

In Detail.controller.ts, read the route parameter and bind the view:

public onInit(): void {
  const oRouter = this.getOwnerComponent()?.getRouter();
  oRouter?.getRoute("detail")?.attachPatternMatched(
    this._onRouteMatched.bind(this)
  );
}

private _onRouteMatched(oEvent: Event): void {
  const sId = oEvent.getParameter("arguments").id as string;
  const oModel = this.getOwnerComponent()?.getModel() as JSONModel;
  const aItems = oModel.getProperty("/items") as Array<{id:string}>;
  const iIndex = aItems.findIndex(i => i.id === sId);

  this.getView()?.bindElement(`/items/${iIndex}`);
}

Test by clicking an item in the list — the detail view should show the item data and the back button should return to the list.

Done! You have implemented the Fiori master-detail navigation pattern. Continue to Exercise 3 to connect to a real OData service.
